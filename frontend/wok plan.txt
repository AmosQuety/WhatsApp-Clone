SQLite Schema Highlights:
-- Messages with selective visibility
CREATE TABLE messages (
  id TEXT PRIMARY KEY,
  group_id TEXT,
  sender_id TEXT,
  text TEXT,
  visible_to TEXT, -- JSON array: ['ALL'] or ['user1', 'user2']
  created_at DATETIME
);
